السطر الكامل مع الشرح:

┌─────────────────────────────────────────────────────────────────────────────┐
│ void BitcoinExchange::processFile(const std::string& fileName)              │
└─────────────────────────────────────────────────────────────────────────────┘

void              ← لا تُرجع أي قيمة
BitcoinExchange:: ← دالة عضو في الفئة BitcoinExchange
processFile       ← اسم الدالة
(                 ← بداية قائمة المعاملات
const             ← البيانات لا يمكن تعديلها داخل الدالة
std::string&      ← مرجع (لا ننسخ السلسلة، نستخدم الأصلية)
fileName          ← اسم المعامل (اسم الملف)
)                 ← نهاية المعاملات

السطر: std::ifstream file(fileName.c_str());
─────────────────────────────────────────
std::ifstream     ← فئة قراءة الملفات
file              ← اسم المتغير
(                 ← بداية المُنشِّئ
fileName.c_str()  ← تحويل std::string إلى C-string (const char*)
)                 ← نهاية المُنشِّئ
;                 ← نهاية التعليمة

السطر: if (!file.is_open())
───────────────────────────
!                 ← عامل النفي (NOT logical)
file.is_open()    ← دالة فحص: هل الملف مفتوح؟
الشرط: إذا كان الملف مغلقاً (NOT open = true)

السطر: std::cerr << "Error: could not open file." << std::endl;
──────────────────────────────────────────────────────
std::cerr         ← مخرجات الأخطاء (error stream)
<<                ← عامل التوجيه (output operator) - يُرسل البيانات
"..."             ← نص الخطأ
std::endl         ← نقل السطر + تفريغ المخزن المؤقت

السطر: size_t divPos = line.find(" | ");
────────────────────────────────────────
size_t            ← نوع بيانات غير موقعي (لتخزين المواضع والأحجام)
divPos            ← اسم المتغير (موضع الفاصل)
line.find(" | ")  ← البحث عن السلسلة " | " في line
                  ← إذا وجدت: تُرجع الموضع (رقم من 0 فما فوق)
                  ← إذا لم توجد: تُرجع std::string::npos

السطر: if (divPos == std::string::npos)
──────────────────────────────────────
==                ← عامل المساواة
std::string::npos ← ثابت يعني "موضع غير موجود"
                  ← عادةً قيمة كبيرة جداً (مثل -1 لكن بدون إشارة)

السطر: std::string date = line.substr(0, divPos);
──────────────────────────────────────────────────
=                 ← إسناد (تعيين القيمة)
line.substr()     ← استخراج جزء من السلسلة
(0, divPos)       ← من الموضع 0 بطول divPos حرف
مثال: "2011-06-15 | 0.5".substr(0, 10) = "2011-06-15"

السطر: std::string value = line.substr(divPos + 3);
──────────────────────────────────────────────────
divPos + 3        ← تخطي " | " (ثلاثة أحرف)
.substr(divPos+3) ← من بعد الفاصل إلى النهاية
مثال: "2011-06-15 | 0.5".substr(13) = "0.5"

السطر: date.erase(0, date.find_first_not_of(" \t"));
────────────────────────────────────────────────
.erase(            ← دالة حذف
0,                 ← من الموضع 0
date.find_first_not_of(" \t")
                  ← عدد الأحرف = موضع أول حرف ليس مسافة أو tab
الغرض: حذف المسافات من البادية

السطر: date.erase(date.find_last_not_of(" \t") + 1);
────────────────────────────────────────────────
find_last_not_of() ← موضع آخر حرف ليس مسافة
+ 1                ← لشمل ذلك الحرف
.erase(            ← حذف من هنا إلى النهاية
الغرض: حذف المسافات من النهاية

مثال كامل:
    "  2011-06-15  " → بعد تنظيف → "2011-06-15"

السطر: double rate = getRateForDate(date);
──────────────────────────────────────────
double            ← نوع البيانات (رقم عشري مزدوج الدقة)
rate              ← اسم المتغير
getRateForDate()  ← استدعاء دالة البحث عن السعر

السطر: double amount = std::strtod(value.c_str(), NULL);
───────────────────────────────────────────────────────
std::strtod()     ← دالة تحويل نص إلى رقم عشري
value.c_str()     ← تحويل std::string إلى const char*
NULL              ← لا نريد معلومات عن بقية النص
مثال: std::strtod("0.5", NULL) → 0.5

السطر: std::cout << date << " => " << amount << " = " << result << std::endl;
──────────────────────────────────────────────────────────────────────────
std::cout         ← مخرجات القياسية (console output)
<<                ← عامل التوجيه (يُرسل بيانات)
مسلسل التوجيهات:
  date            → "2011-06-15"
  " => "          → نص ثابت
  amount          → 0.5
  " = "           → نص ثابت
  result          → 12.44
  std::endl       → نقل سطر

النتيجة على الشاشة:
  2011-06-15 => 0.50 = 12.44

📌 دالة معقدة أخرى: getRateForDate()

المنطق المعقد:

السطر: std::map<std::string, double>::const_iterator it;
──────────────────────────────────────────────────────
std::map<std::string, double>
                  ← نوع البيانات الكامل
::const_iterator  ← مؤشر ثابت (لا يمكن تعديل العناصر)
it                ← اسم المتغير (iterator = مؤشر)

السطر: it = rates.find(date);
───────────────────────────
rates             ← std::map تخزن البيانات
.find(date)       ← البحث عن المفتاح (date)
                  ← ترجع:
                     - إذا وجد: مؤشر يشير للعنصر
                     - إذا لم يوجد: مؤشر يشير إلى rates.end()

السطر: if (it != rates.end())
───────────────────────────
!=                ← عامل عدم المساواة
rates.end()       ← النهاية (عنصر خيالي بعد آخر عنصر)
الشرط: إذا وجد العنصر (الحاوية ليست في النهاية)
└─ return (it->second); ← إرجاع القيمة (السعر)

السطر: it = rates.lower_bound(date);
────────────────────────────────────
lower_bound()     ← أول عنصر بمفتاح >= المطلوب
مثال (البحث عن "2011-06-20"):
    الخريطة:
    | "2011-06-10" → 10 |
    | "2011-06-15" → 15 |  ← lower_bound يشير هنا
    | "2011-06-25" → 25 |
    
    لأن "2011-06-20" < "2011-06-25"

السطر: if (it == rates.begin())
───────────────────────────────
rates.begin()     ← أول عنصر في الخريطة
==                ← هل المؤشر في البداية؟
إذا كان في البداية: لا يوجد عنصر سابق
└─ return (-1);    ← خطأ (لا توجد بيانات قديمة)

السطر: --it;
──────────
--                ← عامل الإنقاص المسبق (decrement)
it                ← تقليل المؤشر بمقدار واحد (الذهاب للعنصر السابق)

السطر: return (it->second);
──────────────────────────
it->              ← الوصول إلى العنصر المؤشر عليه
second            ← حقل القيمة (الجزء الثاني من الزوج)

مثال كامل:
┌─────────────────────────────────────────────────────────┐
│ البحث عن سعر: 2011-06-20 (تاريخ غير موجود)           │
│                                                         │
│ الخريطة:                                               │
│ | "2011-06-10" → 10.0  |                               │
│ | "2011-06-15" → 15.2  | ← `it` بعد lower_bound()    │
│ | "2011-06-25" → 25.8  |                               │
│                                                         │
│ العمليات:                                             │
│ 1. it.find("2011-06-20") → يشير إلى النهاية (لم يجد)  │
│ 2. it.lower_bound("2011-06-20") → يشير إلى الموضع الذي│
│                                    سيُدرج فيه الميدل  │
│                                    (أي "2011-06-25")  │
│ 3. it != begin() ✓                                     │
│ 4. --it → يشير إلى "2011-06-15"                       │
│ 5. return 15.2                                         │
└─────────────────────────────────────────────────────────┘

السطر: if (date.length() != 10)
────────────────────────────
.length()         ← عدد الأحرف في السلسلة
!=                ← عدم المساواة
10                ← الطول المتوقع (YYYY-MM-DD = 10)
الغرض: تصفية الصيغ الخاطئة مبكراً

السطر: if (date[4] != '-' || date[7] != '-')
──────────────────────────────────────────
date[4]           ← الحرف في الموضع 4 (الخامس)
                  ← في "2011-06-15": [4] = '-'
||                ← عامل OR (أو)
                  ← إذا كان أحد الشرطين صحيح → true
الغرض: التحقق من وجود الفواصل في المواضع الصحيحة

السطر: for (int i = 0; i < 10; i++)
──────────────────────────────────
int i = 0;        ← متغير العداد من 0
i < 10;           ← الشرط (ما دام أقل من 10)
i++               ← إضافة 1 بعد كل تكرار
الحلقة: تكرار 10 مرات (i من 0 إلى 9)

السطر: if (i == 4 || i == 7) continue;
──────────────────────────────────────
==                ← عامل المساواة
||                ← أو
continue;         ← انتقل للدورة التالية (تخطي البقية)
الغرض: تجاهل فحص الفواصل (نعلم أنها صحيحة)

السطر: if (!std::isdigit(date[i])) return (false);
──────────────────────────────────────────────────
std::isdigit()    ← دالة من مكتبة <cctype>
                  ← ترجع true إذا كان الحرف رقم (0-9)
!                 ← نفي (إذا لم يكن رقم)
return (false);   ← إرجاع false على الفور
الغرض: التأكد من أن كل حرف (غير الفواصل) هو رقم

السطر: int year = std::strtol(date.substr(0, 4).c_str(), NULL, 10);
──────────────────────────────────────────────────────────────────
date.substr(0, 4) ← الأحرف من 0 إلى 4 = السنة (مثل "2011")
.c_str()          ← تحويل إلى C-string
std::strtol()     ← تحويل إلى رقم صحيح طويل (long)
(   , NULL, 10)   ← NULL: لا نهتم ببقية النص
                  ← 10: نظام عشري
int year          ← تحويل من long إلى int
النتيجة: 2011

مثال:
Input:  "2011-06-15"
Output: year = 2011, month = 6, day = 15

السطر: if (year < 2009 || year > 2024) return (false);
─────────────────────────────────────────────────────
<                 ← أقل من
>                 ← أكبر من
||                ← أو
الشرط: إذا كانت السنة خارج النطاق (2009-2024)
الغرض: تصفية السنوات غير المعقولة

السطر: char *endptr;
──────────────────
char *            ← مؤشر على حرف
endptr            ← سيشير إلى الموضع الذي توقف البحث فيه
الغرض: كشف الأحرف الزائدة بعد الرقم

السطر: double num = std::strtod(value.c_str(), &endptr);
───────────────────────────────────────────────────────
std::strtod()     ← تحويل نص إلى double
value.c_str()     ← النص (C-string)
&endptr           ← عنوان المؤشر (حتى تملأه الدالة)
النتيجة:
  num = القيمة الرقمية
  endptr = مؤشر للحرف الأول الذي لم يُتحول

مثال:
value = "12.5xyz"
بعد strtod:
  num = 12.5
  endptr → يشير إلى 'x'

السطر: if (*endptr != '\0')
──────────────────────────
*endptr           ← الحرف المشار إليه من endptr
'\0'              ← حرف نهاية السلسلة (null terminator)
!=                ← إذا كان مختلف
الغرض: التحقق من أن نهاية النص (لا توجد أحرف إضافية)

إذا كان endptr يشير إلى 'x' (char != '\0') → خطأ

السطر: if (num < 0 || num > 1000) return (false);
──────────────────────────────────────────────
<                 ← أصغر من
>                 ← أكبر من
||                ← أو
الغرض: تصفية القيم السالبة والكبيرة جداً

السطر: while (std::getline(file, line))
────────────────────────────────────────
while             ← حلقة شرطية (تكرار ما دام الشرط صحيح)
std::getline()    ← دالة قراءة سطر كامل
file              ← مصدر البيانات

───────────────────────────────────
std::istringstream ← فئة قراءة من سلسلة نصية

الغرض: معالجة السطر الواحد

السطر: if (std::getline(iss, date, ',') && std::getline(iss, rateStr))
──────────────────────────────────────────────────────────────────────

                  ← قراءة من iss إلى date حتى ','
&&                ← و (both must be true)
std::getline(iss, rateStr)
                  ← قراءة ما تبقى إلى rateStr
الغرض: فحص أن كلا الجزأين موجود

مثال:
line = "2011-06-15,24.88"
بعد getline:
  date = "2011-06-15"
  rateStr = "24.88"

rateStr.c_str()   ← تحويل إلى C-string
std::strtod()     ← تحويل النص إلى double
NULL              ← لا نهتم بالأخطاء
النتيجة: rate = 24.88

السطر: rates[date] = rate;
───────────────────────────
rates             ← std::map (الخريطة)
[date]            ← الوصول أو الإنشاء بمفتاح date
= rate            ← تعيين القيمة
الغرض: إضافة زوج (تاريخ، سعر) إلى الخريطة

إذا كان المفتاح موجود: تحديث القيمة

📌 الفروقات بين std::string و C strings
================================================================================

C String:         const char* = "hello"
                  ← مصفوفة أحرف منتهية بـ '\0'
                  ← يجب تتبع الطول يدوياً

std::string:      std::string = "hello"
                  ← كائن يدير الذاكرة تلقائياً
                  ← يتتبع الطول بنفسه
                  ← له دوال مفيدة (find, substr, erase, etc)

التحويل:
  std::string → C string: str.c_str()
  C string → std::string: std::string(c_str)

مثال:
    std::string s = "hello";
    const char* c = s.c_str();  ← الحصول على C string
    std::cout << c;             ← طباعة

================================================================================
📌 std::map شرح تفصيلي
================================================================================

تعريف:  std::map<Key, Value>
├─ Key: نوع المفتاح (يجب أن يكون مرتب)
└─ Value: نوع القيمة

في هذا البرنامج:  std::map<std::string, double> rates
├─ Key: std::string (التاريخ) - مرتب أبجدياً
└─ Value: double (السعر)

العمليات الأساسية:
┌──────────────────┬────────────────────────────────────┐
│ الكود             │ الفائدة                            │
├──────────────────┼────────────────────────────────────┤
│ map[key]         │ الوصول أو الإنشاء                │
│ map.at(key)      │ الوصول (خطأ إذا لم يوجد)         │
│ map.find(key)    │ البحث (iterator أو end())        │
│ map.insert()     │ إدراج زوج                         │
│ map.erase()      │ حذف عنصر                          │
│ map.clear()      │ حذف كل العناصر                   │
│ map.size()       │ عدد العناصر                        │
│ map.empty()      │ هل فارغة؟                         │
│ map.begin()      │ المؤشر للأول                      │
│ map.end()        │ المؤشر للنهاية (خيالي)           │
└──────────────────┴────────────────────────────────────┘

الترتيب:
std::map يحافظ على الترتيب الأبجدي للمفاتح
مثال:
    rates["2020-01-01"] = 10;
    rates["2010-01-01"] = 5;
    rates["2015-01-01"] = 7;
    
    عند التكرار:
    → "2010-01-01" (5)
    → "2015-01-01" (7)
    → "2020-01-01" (10)

البحث:
    auto it = rates.find("2015-01-01");
    if (it != rates.end()) {
        std::cout << it->second;  ← 7
    }

lower_bound:
    it = rates.lower_bound("2012-06-15");
    → المؤشر الأول بقيمة >= "2012-06-15"

upper_bound:
    it = rates.upper_bound("2012-06-15");
    → المؤشر الأول بقيمة > "2012-06-15"
