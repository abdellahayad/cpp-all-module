🎯 ما يفعل البرنامج
════════════════════════════════════════════════════════════════════════════

البرنامج يحسب التعبيرات الرياضية المكتوبة بصيغة بولندية عكسية (RPN).

الصيغة الطبيعية (Infix):    8 + 9
الصيغة البولندية العكسية:  8 9 +

مثال كامل:
┌──────────────────────────────────────────┐
│ Infix (الصيغة الطبيعية):                │
│ (8 * 9) - 9 - 9 - 9 - 4 + 1              │
│                                          │
│ RPN (الصيغة البولندية العكسية):        │
│ 8 9 * 9 - 9 - 9 - 4 - 1 +               │
│                                          │
│ النتيجة: 42                             │
└──────────────────────────────────────────┘

════════════════════════════════════════════════════════════════════════════
💡 كيف تعمل صيغة RPN
════════════════════════════════════════════════════════════════════════════

تُستخدم stack (مكدس) لمعالجة التعبير:

1. اقرأ كل رمز (token) من التعبير من اليسار إلى اليمين
2. إذا كان رقماً: أدخله في الـ stack
3. إذا كان عاملاً (+, -, *, /):
   a. أخرج رقمين من الـ stack
   b. طبّق العملية
   c. أدخل النتيجة في الـ stack
4. النتيجة النهائية هي الرقم الوحيد المتبقي في الـ stack

مثال: "3 4 +" (الجمع: 3 + 4)
┌─────────────────────────────────┐
│ الخطوة 1: اقرأ 3 → أدخل في stack  │
│ Stack: [3]                      │
│                                 │
│ الخطوة 2: اقرأ 4 → أدخل في stack  │
│ Stack: [3, 4]                   │
│                                 │
│ الخطوة 3: اقرأ + → عملية!         │
│ - أخرج 4 (b)                   │
│ - أخرج 3 (a)                   │
│ - احسب a + b = 3 + 4 = 7       │
│ - أدخل 7 في stack              │
│ Stack: [7]                      │
│                                 │
│ النتيجة: 7                     │
└─────────────────────────────────┘

مثال معقد: "8 9 * 9 - 9 - 9 - 4 - 1 +"

┌──────────────────────────────────────────┐
│ Step 1: اقرأ 8 → Stack: [8]              │
│ Step 2: اقرأ 9 → Stack: [8, 9]           │
│ Step 3: اقرأ * → احسب 8 * 9 = 72        │
│         Stack: [72]                      │
│ Step 4: اقرأ 9 → Stack: [72, 9]          │
│ Step 5: اقرأ - → احسب 72 - 9 = 63       │
│         Stack: [63]                      │
│ Step 6: اقرأ 9 → Stack: [63, 9]          │
│ Step 7: اقرأ - → احسب 63 - 9 = 54       │
│         Stack: [54]                      │
│ Step 8: اقرأ 9 → Stack: [54, 9]          │
│ Step 9: اقرأ - → احسب 54 - 9 = 45       │
│         Stack: [45]                      │
│ Step 10: اقرأ 4 → Stack: [45, 4]         │
│ Step 11: اقرأ - → احسب 45 - 4 = 41      │
│          Stack: [41]                     │
│ Step 12: اقرأ 1 → Stack: [41, 1]         │
│ Step 13: اقرأ + → احسب 41 + 1 = 42      │
│          Stack: [42]                     │
│                                          │
│ النتيجة النهائية: 42                   │
└──────────────────────────────────────────┘

🔍 شرح ملف RPN.hpp (الرأس)
════════════════════════════════════════════════════════════════════════════

    #include <string>       ← نوع البيانات string
    #include <deque>        ← الحاوية deque (مكدس محتمل)
    #include <sstream>      ← لتحليل النصوص
    #include <cmath>        ← العمليات الرياضية
    #include <cstdlib>      ← للتحويل strtod
    #include <iostream>     ← المدخلات والمخرجات
    #include <stdexcept>    ← للاستثناءات

    class RPN { ... }

    private:
        std::deque<double> stack;
                            ↑
    الحاوية الرئيسية: deque (double-ended queue)
    - يمكن إضافة وحذف من الطرفين بسرعة O(1)
    - نستخدمها كـ Stack (عمليات من الطرف الخلفي فقط):
      * push_back() - إضافة
      * pop_back() - حذف
      * back() - الوصول للعنصر الأخير

    معلومة: في ex00 استخدمنا map، لذا محظور هنا!

🔍 شرح ملف RPN.cpp (التطبيق)
════════════════════════════════════════════════════════════════════════════

▶ فحص إذا كان النص رقماً:
─────────────────────────────────────────────────────────────────────────
    bool RPN::isNumber(const std::string& token) const
    {
        if (token.empty())          ← إذا فارغ، ليس رقم
            return (false);
        
        for (size_t i = 0; i < token.length(); i++)
        {
            if (!std::isdigit(token[i]))  ← تحقق كل حرف
                return (false);
        }
        
        return (true);              ← كل شيء صحيح
    }

    مثل: "123" → true
         "abc" → false
         "12a" → false

▶ فحص إذا كان النص عاملاً:
─────────────────────────────────────────────────────────────────────────
    bool RPN::isOperator(const std::string& token) const
    {
        if (token.length() != 1)    ← يجب أن يكون حرف واحد فقط
            return (false);
        
        char c = token[0];
        return (c == '+' || c == '-' || c == '*' || c == '/');
                                    ← تحقق من العاملات المدعومة
    }

    مثل: "+" → true
         "-" → true
         "*" → true
         "/" → true
         "++" → false (أكثر من حرف)
         "a" → false (ليس عاملاً)

▶ تطبيق العملية الحسابية:
─────────────────────────────────────────────────────────────────────────
    double RPN::applyOperation(double a, double b, char op) const
    {
        switch (op)
        {
            case '+': return (a + b);   ← الجمع
            case '-': return (a - b);   ← الطرح
            case '*': return (a * b);   ← الضرب
            case '/':
                if (b == 0)             ← تحقق من القسمة على صفر
                    throw std::runtime_error("Error: division by zero");
                return (a / b);         ← القسمة
            default:
                throw std::runtime_error("Error: unknown operator");
        }
    }

    ملاحظة المرتبة: الرقم الأول (a) يُخرج أولاً، الثاني (b) يُخرج ثانياً
    إذا كان الـ stack: [3, 5] والعامل: -
    نخرج 5 (b)، ثم 3 (a)
    النتيجة: a - b = 3 - 5 = -2 ✓

▶ دالة حساب التعبير (الأهم):
─────────────────────────────────────────────────────────────────────────
    int RPN::calculate(const std::string& expression)
    {
        stack.clear();  ← تفريغ الـ stack من البيانات السابقة
        
        std::istringstream iss(expression);  ← فتح بث للنص
        std::string token;                    ← متغير لكل رمز
        
        while (iss >> token)  ← قراءة كل رمز (مفصول بمسافة)
        {
            if (isNumber(token))  ← إذا كان رقم
            {
                double number = strtod(token.c_str(), NULL);
                                  ← تحويل النص إلى رقم
                stack.push_back(number);  ← أضف إلى الـ stack
            }
            else if (isOperator(token))  ← إذا كان عامل
            {
                if (stack.size() < 2)  ← نحتاج رقمين على الأقل
                    throw std::runtime_error(
                        "Error: insufficient operands");
                
                // خرج الشكل الصحيح للرقمين:
                double b = stack.back();    ← الرقم الثاني
                stack.pop_back();
                
                double a = stack.back();    ← الرقم الأول
                stack.pop_back();
                
                // طبق العملية
                double result = applyOperation(a, b, token[0]);
                
                // ادخل النتيجة
                stack.push_back(result);
            }
            else  ← رمز غير معروف
            {
                throw std::runtime_error("Error: invalid token");
            }
        }
        
        if (stack.size() != 1)  ← من المفروض رقم واحد فقط
            throw std::runtime_error("Error: invalid expression");
        
        return (static_cast<int>(stack.back()));  ← إرجاع النتيجة
    }

📊 مثال تفصيلي: حساب "3 4 + 2 *"
════════════════════════════════════════════════════════════════════════════

التعبير: "3 4 + 2 *" = "(3 + 4) * 2" = 14

الخطوات:

1️⃣  اقرأ "3"
    - isNumber("3") → true
    - أدخل 3
    - Stack: [3]

2️⃣  اقرأ "4"
    - isNumber("4") → true
    - أدخل 4
    - Stack: [3, 4]

3️⃣  اقرأ "+"
    - isOperator("+") → true
    - stack.size() = 2 ✓
    - b = 4 (pop)
    - a = 3 (pop)
    - result = applyOperation(3, 4, '+') = 3 + 4 = 7
    - أدخل 7
    - Stack: [7]

4️⃣  اقرأ "2"
    - isNumber("2") → true
    - أدخل 2
    - Stack: [7, 2]

5️⃣  اقرأ "*"
    - isOperator("*") → true
    - stack.size() = 2 ✓
    - b = 2 (pop)
    - a = 7 (pop)
    - result = applyOperation(7, 2, '*') = 7 * 2 = 14
    - أدخل 14
    - Stack: [14]

6️⃣  نهاية الحلقة
    - stack.size() = 1 ✓
    - إرجاع 14

    $ ./RPN "8 9 * 9 - 9 - 9 - 4 - 1 +"

    42

    0

    $ ./RPN "(1 + 1)"
    Error: invalid token

    Error: insufficient operands

    $ ./RPN "3 4 + 5"
    Error: invalid expression

    $ ./RPN "5 0 /"
